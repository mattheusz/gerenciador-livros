create table Livro(
    id_livro integer(4) not null primary key AUTO_INCREMENT,
    foto varchar(80),
    titulo varchar(80) not null,  
    descricao varchar(200),
    autor varchar(80),
    editora varchar(80),
    isbn varchar (13),   
paginas numeric(4),
    pag_lidas numeric(4),
    adquirido varchar(5));



alter table Livro
  add constraint Livro_ibns_UK
  unique key (isbn);
